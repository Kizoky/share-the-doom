class PostM16 : PostalWeapon replaces Chaingun
{
	Default
	{
    Weapon.Kickback 100;
	Decal "bulletchip";
    Weapon.SelectionOrder 1300;
	Weapon.AmmoUse 1;
	Weapon.AmmoGive 8;
	Weapon.AmmoType "Clip";
	Obituary "$OB_MACHINEGUN";
    Tag "$TAG_MACHINEGUN";
	Inventory.PickupMessage "$GOTMACHINEGUN";
	}
	States
	{
	Ready:
		TNT1 A 1
		{
			cvar idlinganimcvar = cvar.findcvar('postal_idleanim');
			bool idlinganim = (idlingAnimCVar != null && idlingAnimCVar.GetBool ());
			
			if (idlinganim)
				return ResolveState ("Ready.IdleAnim");
			else
				return ResolveState ("Ready.NoIdleAnim");
			
			return ResolveState (null);
		}
		Loop;
	Ready.IdleAnim:
		MGID ABCDEFGHIJKLMNOPQRSTU 7 A_WeaponReady;
		Goto Ready;
	Ready.NoIdleAnim:
		MGID A 1 A_WeaponReady;
		Goto Ready;
	Deselect:
		TNT1 AAAAAAAAAAAAAAA 0 A_Lower;
		MGDT AA 1;
		Loop;
	Select:
		TNT1 AAAAAAAAAAAAAAA 0 A_Raise;
		TNT1 A 0 A_WeaponOffset(0,32); // Fix "bump" when weapon is selected
		TNT1 A 0 A_PlaySound("weaponselect2", 7 | 4096);
		MGST ABCDEFGHIJKLMNOPQQ 2;
		Loop;
	Fire:
		MGFE A 0 A_SetPitch(pitch-0.50);
		MGFE A 5 { A_FireCGun(); } // ABCDEFGHIJKLMNOP
		MGFE NOP 1 A_ReFire;
		MGID A 0 A_SetPitch(pitch+0.50);
		Goto Ready;
	Flash:
		//PGNS B 3 Offset(0, 0) Bright A_Light2;
		//PGNS A 2 Offset(0, 0) Bright A_Light1;
		Goto LightDone;
	Spawn:
		TNT1 A 0;
		PICK A -1;
		Stop;
	}
}
