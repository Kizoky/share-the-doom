class PostMatchbox : PostalWeapon
{
	Default
	{
    Weapon.Kickback 100;
    Weapon.SelectionOrder 1300;
	Obituary "$OB_MPSHOTGUN";
    Tag "$TAG_SHOTGUN";
	Inventory.PickupMessage "$GOTSHOTGUN";
	+Weapon.NoAlert;
	}
	States
	{
	Ready:
		TNT1 A 1
		{
			cvar idlinganimcvar = cvar.findcvar('postal_idleanim');
			bool idlinganim = (idlingAnimCVar != null && idlingAnimCVar.GetBool ());
			
			if (idlinganim)
				return ResolveState ("Ready.IdleAnim");
			else
				return ResolveState ("Ready.NoIdleAnim");
			
			return ResolveState (null);
		}
		Loop;
	Ready.IdleAnim:
		MATI ABCDEFGHIJKLMNOP 7 A_WeaponReady;
		MATI PONMLKJIHGFEDCBA 0;
		Goto Ready;
	Ready.NoIdleAnim:
		MATI A 1 A_WeaponReady;
		Goto Ready;
	Deselect:
		TNT1 AAAAAAAAAAAAAAA 0 A_Lower;
		MATD ABCDEFGHIJKL 2;
		Loop;
	Select:
		TNT1 AAAAAAAAAAAAAAA 0 A_Raise;
		TNT1 A 0 A_WeaponOffset(0,32); // Fix "bump" when weapon is selected
		TNT1 A 0 A_PlaySound("weaponselect1", 7 | 4096);
		MATS ABCDEFGHIJKLMNO 2;
		Loop;
	Fire:
		MA1F ABCDEFGHIJKLMNOPQRSTUVWXYZ 1;
		MA2F ABCDEFGHIJKLMNOPQRSTUVWXYZ 1;
		Goto Ready;
	Spawn:
		PICK A -1;
		Stop;
	}
}
