class PostWMD : PostalWeapon
{
	Default
	{
		//$Category STD Weapons
		//$Title "W.M.D."
		//$Sprite PICKA0
		Weapon.Kickback 100;
		Weapon.SelectionOrder 1300;
		Weapon.BobRangeX 0.4;
		Weapon.BobRangeY 0.5;
		Obituary "$PSTL_OB_WMD";
		Tag "$PSTL_WP_WMD";
		Inventory.PickupMessage "$PSTL_PP_WMD";
		+Weapon.NoAlert;
	}
	States
	{
	Ready:
		TNT1 A 1
		{
			cvar idlinganimcvar = cvar.findcvar('postal_idleanim');
			bool idlinganim = (idlingAnimCVar != null && idlingAnimCVar.GetBool ());
			
			if (idlinganim)
				return ResolveState ("Ready.IdleAnim");
			else
				return ResolveState ("Ready.NoIdleAnim");
			
			return ResolveState (null);
		}
		Loop;
	Ready.IdleAnim:
		WMDS A 9 A_WeaponReady;
		Goto Ready;
	Ready.NoIdleAnim:
		WMDS A 1 A_WeaponReady;
		Goto Ready;
	Deselect:
		WMDS A 2 Offset(0,0) A_Lower;
		Loop;
	Select:
		TNT1 A 0 A_PlaySound("weaponselect1", 7 | 4096);
		WMDS A 2 A_Raise;
		Loop;
	Fire:
		WMDS A 5 A_FireShotgun;
		Goto Ready;
	Spawn:
		TNT1 A 0;
		PICK A -1;
		Stop;
	}
}
