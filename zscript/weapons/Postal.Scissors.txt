class PostScissors : PostalWeapon
{
	Default
	{
    Weapon.Kickback 100;
    Weapon.SelectionOrder 1300;
	Obituary "$OB_MPSHOTGUN";
    Tag "$TAG_SHOTGUN";
	Inventory.PickupMessage "$GOTSHOTGUN";
	+Weapon.NoAlert;
	}
	States
	{
	Ready:
		TNT1 A 1
		{
			cvar idlinganimcvar = cvar.findcvar('postal_idleanim');
			bool idlinganim = (idlingAnimCVar != null && idlingAnimCVar.GetBool ());
			
			if (idlinganim)
				return ResolveState ("Ready.IdleAnim");
			else
				return ResolveState ("Ready.NoIdleAnim");
			
			return ResolveState (null);
		}
		Loop;
	Ready.IdleAnim:
		SCII ABCDEFGHIJKLMNOPQRSTUVWXYZ 3 A_WeaponReady;
		Goto Ready;
	Ready.NoIdleAnim:
		SCII A 1 A_WeaponReady;
		Goto Ready;
	Deselect:
		TNT1 AAAAAAAAAAAAAAA 0 A_Lower;
		SCID ABCDEFGHIJKLMNO 2;
		Loop;
	Select:
		TNT1 AAAAAAAAAAAAAAA 0 A_Raise;
		TNT1 A 0 A_WeaponOffset(0,32); // Fix "bump" when weapon is selected
		TNT1 A 0 A_PlaySound("weaponselect1", 7 | 4096);
		SCIS ABCDEFGHIJKLMNOPQRS 2 A_WeaponReady;
		Loop;
	Fire:
		SCIF ABCDEFGHIJKL 2 A_FireShotgun;
		Goto Ready;
	AltFire:
		SCSF ABCDEFGHIJKLMNOPQRST 2 A_FireShotgun;
		Goto Ready;
	Spawn:
		TNT1 A 0;
		PICK A -1;
		Stop;
	}
}
