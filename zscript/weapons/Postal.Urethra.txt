class PostUrethra : PostalWeapon
{
	bool Sigh;
	
	override void Tick()
	{
		Super.Tick();
		
		if (!Owner) return;
	
	}
	
	Default
	{
		Weapon.Kickback 100;
		Weapon.SelectionOrder 1300;
		Weapon.AmmoUse 0;
		Weapon.AmmoType "Piss";
		Obituary "$PSTL_OB_URETHRA";
		Tag "$PSTL_WP_URETHRA";
		Inventory.PickupMessage "$PSTL_PP_URETHRA";
		+Weapon.NoAlert;
	}
	States
	{
	Ready:
		URET A 1 A_WeaponReady;
		Goto Ready;
	Deselect:
		URET A 0 A_StartSound("piss/zipperup", CHAN_BODY);
		URET A 1 A_Lower;
		Loop;
	Select:
		URET A 0 A_StartSound("piss/zipperdown", CHAN_BODY);
		URET A 1 A_Raise;
		Loop;
	Fire:
		URET A 0
		{
			if (!invoker.Sigh) invoker.Sigh = true;
		}
		URET A 1 A_StartSound("piss/loop", CHAN_WEAPON);
		Goto Ready;
	Spawn:
		SHOT A -1;
		Stop;
	}
}

class Urine : Actor
{
	override void PostBeginPlay()
	{
		Super.PostBeginPlay();
		
		int r = random(1,2);
		switch(r)
		{
			case 1:
				SetStateLabel("One");
				break;
			case 2:
				SetStateLabel("Two");
				break;
			default:
				break;
		}
	
	}
	
	Default
	{
		Radius 5;
		Height 5;
		Speed 5;
		Damage 0;
		Projectile;
		+RANDOMIZE
		-NOBLOCKMAP
		-NOGRAVITY
		+SHOOTABLE
		+NOBLOOD
		Gravity 0.27;
		BloodType "null";
		//SeeSound "";
		DeathSound "";
		//BounceSound "";
		Obituary "%o got pissed on by %k ... ???";
	}
	States
	{
		Spawn:
			TNT1 A 0;
			Loop;
		One:
			URIN A 3;
			Loop;
		Two:
			URIN B 3;
			Loop;
		Death:
			TNT1 A 0
			{
				Destroy();
			}
			Stop;
	}
}
