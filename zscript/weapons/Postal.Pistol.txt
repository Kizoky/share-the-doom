class PostPistol : PostalWeapon replaces Pistol
{
	Default
	{
	Weapon.Kickback 100;
	Decal "bulletchip";
	Weapon.SelectionOrder 1900;
	Weapon.AmmoUse 1;
	Weapon.AmmoGive 20;
	Weapon.AmmoType "Clip";
	Obituary "$OB_MPPISTOL";
	//+WEAPON.WIMPY_WEAPON;
	Tag "$TAG_PISTOL";
	Inventory.PickupMessage "$GOTPISTOL";
	}
	States
	{
	Ready:
		TNT1 A 1
		{
			cvar idlinganimcvar = cvar.findcvar('postal_idleanim');
			bool idlinganim = (idlingAnimCVar != null && idlingAnimCVar.GetBool ());
			
			if (idlinganim)
				return ResolveState ("Ready.IdleAnim");
			else
				return ResolveState ("Ready.NoIdleAnim");
			
			return ResolveState (null);
		}
	Ready.IdleAnim:
		POXX ABCDEFGHIJKLMNOPQRSTUVWXYZ 7 A_WeaponReady;
		Goto Ready;
	Ready.NoIdleAnim:
		POXX A 1 A_WeaponReady;
		Goto Ready;
	Deselect:
		TNT1 AAAAAAAAAAAAAAA 0 A_Lower;
		PXXX ABCDEFGHIJKLMNOPQRSTUVWXX 1;
		Loop;
	Select:
		TNT1 AAAAAAAAAAAAAAA 0 A_Raise;
		TNT1 A 0 A_WeaponOffset(0,32); // Fix "bump" when weapon is selected
		TNT1 A 0 A_PlaySound("weaponselect3", 7 | 4096);
		POPS ABCDEFGHIJKLMNOPQRSTUVWXYZ 1 A_WeaponReady;
		Loop;
	Fire:
		POXA AB 1;
		//TNT1 A 0 A_QuakeEx(1, 1, 1, 7, 0, 60 | QF_RELATIVE);
		POXA C 0 A_SetPitch(pitch-0.50);
		POXA C 2 A_FirePistol;
		POXA D 0 A_SetPitch(pitch+0.50);
		POXA DEFGHIJKLM 1;
		POXA STUVWXYZ 2;
		//POX2 ABCD 4 A_ReFire;
		POX2 A 1 A_ReFire;
		Goto Ready;
	Spawn:
		TNT1 A 0;
		PICK A -1;
		Stop;
	}
}
