class PostHands : PostalWeapon
{
	Default
	{
	Weapon.Kickback 100;
	Weapon.SelectionOrder 9999;
	Obituary "$PSTL_OB_HANDS";
	Tag "$PSTL_WP_HANDS";
	Inventory.PickupMessage "$PSTL_PP_HANDS";
	+Weapon.NoAlert;
	}
	States
	{
	Ready:
		ABAB A 1 A_WeaponReady;
		Goto Ready;
	Deselect:
		ABAB A 1 A_Lower;
		Loop;
	Select:
		ABAB A 1 A_Raise;
		Loop;
	Fire:
		Goto Ready;
	}
}
