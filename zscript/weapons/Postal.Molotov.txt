class PostMolotov : PostalWeapon
{
	Default
	{
		//$Category STD Weapons
		//$Title "Molotov"
		//$Sprite PICKA0
		Weapon.Kickback 100;
		Weapon.SelectionOrder 1300;
		Obituary "$OB_MPSHOTGUN";
		Tag "$TAG_SHOTGUN";
		Inventory.PickupMessage "Picked up a molotov.";
		+Weapon.NoAlert;
		Obituary "%o just enjoyed a Tibetan Monk experience thanks to %k's Molotov.";
	}
	States
	{
	Ready:
		TNT1 A 1
		{
			cvar idlinganimcvar = cvar.findcvar('postal_idleanim');
			bool idlinganim = (idlingAnimCVar != null && idlingAnimCVar.GetBool ());
			
			if (idlinganim)
				return ResolveState ("Ready.IdleAnim");
			else
				return ResolveState ("Ready.NoIdleAnim");
			
			return ResolveState (null);
		}
		Loop;
	Ready.IdleAnim:
		MOLI ABCDEFGHIJKLMNOPQRS 7 A_WeaponReady;
		Goto Ready;
	Ready.NoIdleAnim:
		MOLI A 1 A_WeaponReady;
		Goto Ready;
	Deselect:
		TNT1 AAAAAAAAAAAAAAA 0 A_Lower;
		MOLD ABCDEFGHIJKLMNOPQRSTUVWX 2;
		Loop;
	Select:
		TNT1 AAAAAAAAAAAAAAA 0 A_Raise;
		TNT1 A 0 A_WeaponOffset(0,32); // Fix "bump" when weapon is selected
		TNT1 A 0 A_PlaySound("weaponselect1", 7 | 4096);
		MOLS ABCDEFGHIJKLMNOPQRSTUVWXYZ 2 A_WeaponReady;
		Loop;
	Fire:
		MOL1 ABCDEFGHIJKLMNOPQRSTUVWXYZ 2 A_FireShotgun;
		MOL2 ABCDE 2;
		Goto Ready;
	AltFire:
		MLSF ABCDEFGHIJKLMNOPQ 2;
		Goto Ready;
	Spawn:
		TNT1 A 0;
		PICK A -1;
		Stop;
	}
}
