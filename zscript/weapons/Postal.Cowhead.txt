class PostCowhead : PostalWeapon
{
	Default
	{
    Weapon.Kickback 100;
    Weapon.SelectionOrder 1300;
	Obituary "$OB_MPSHOTGUN";
    Tag "$TAG_SHOTGUN";
	Inventory.PickupMessage "$GOTSHOTGUN";
	+Weapon.NoAlert;
	}
	States
	{
	Ready:
		TNT1 A 1
		{
			cvar idlinganimcvar = cvar.findcvar('postal_idleanim');
			bool idlinganim = (idlingAnimCVar != null && idlingAnimCVar.GetBool ());
			
			if (idlinganim)
				return ResolveState ("Ready.IdleAnim");
			else
				return ResolveState ("Ready.NoIdleAnim");
			
			return ResolveState (null);
		}
		Loop;
	Ready.IdleAnim:
		COWI ABCDEFGHIJKLMNOPQRSTUVWXYZ 5 A_WeaponReady;
		Goto Ready;
	Ready.NoIdleAnim:
		COWI A 1 A_WeaponReady;
		Goto Ready;
	Deselect:
		COWD ABCDEFGHIJKLMNOPQRSTUVWXYZ 2 A_Lower;
		Loop;
	Select:
		TNT1 A 0 A_PlaySound("weaponselect1", 7 | 4096);
		COWS ABCDEFGHIJKLMNOPQRSTUV 2 A_Raise;
		Loop;
	Fire:
		COWF ABCDEFGHIJKLMNOPQRSTUVWXY 3 A_FireShotgun;
		Goto Ready;
	AltFire:
		COSF ABCDEFGHIJKLMNOPQRS 3 A_FireShotgun;
		Goto Ready;
	Spawn:
		TNT1 A 0;
		PICK A -1;
		Stop;
	}
}
