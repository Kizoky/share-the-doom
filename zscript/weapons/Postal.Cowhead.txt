class PostCowhead : PostalWeapon
{
	Default
	{
		//$Category STD Weapons
		//$Title "Severed Cowhead"
		//$Sprite PICKA0
		Weapon.Kickback 100;
		Weapon.SelectionOrder 1300;
		Obituary "$PSTL_OB_COWHEAD";
		Tag "$PSTL_WP_COWHEAD";
		Inventory.PickupMessage "$PSTL_PP_COWHEAD";
		+Weapon.NoAlert;
	}
	States
	{
	Ready:
		COWI ABCDEFGHIJKLMNOPQRSTUVWXYZ 5 A_WeaponReady;
		Loop;
	Deselect:
		COWD ABCDEFGHIJKLMNOPQRSTUVWXYZ 2 A_Lower;
		Loop;
	Select:
		COWS A 0 A_PostalRaise(0);
		COWS A 0 A_PlaySound("weaponselect1", 7 | 4096);
		COWS ABCDEFGHIJKLMNOPQRSTUV 2 A_WeaponReady;
		COWI A 0 A_PostalRaise(1);
		Loop;
	Fire:
		COWF ABCDEFGHIJKLMNOPQRSTUVWXY 3 A_FireShotgun;
		Goto Ready;
	AltFire:
		COSF ABCDEFGHIJKLMNOPQRS 3 A_FireShotgun;
		Goto Ready;
	Spawn:
		TNT1 A 0;
		PICK A -1;
		Stop;
	}
}
