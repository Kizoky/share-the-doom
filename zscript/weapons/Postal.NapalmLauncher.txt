class PostNapalmLauncher : PostalWeapon
{
	Default
	{
		//$Category STD Weapons
		//$Title "Napalm Launcher"
		//$Sprite PICKA0
		Weapon.Kickback 100;
		Weapon.SelectionOrder 1300;
		Obituary "$PSTL_OB_NAPALM";
		Tag "$PSTL_WP_NAPALM";
		Inventory.PickupMessage "$PSTL_PP_NAPALM";
		+Weapon.NoAlert;
	}
	States
	{
	Ready:
		NAPI ABCDEFGHIJKLMNOPQRS 7 A_WeaponReady;
		Loop;
	Deselect:
		NAPD ABCDEFGHI 2;
		NAPD I 0 A_PostalLower();
		Loop;
	Select:
		NAPS A 0 A_PostalRaise(0);
		NAPS A 0 A_PlaySound("weaponselect1", 7 | 4096);
		NAPS ABCDEFGHIJKLMNOPQ 2 A_WeaponReady;
		NAPI A 0 A_PostalRaise(1);
		Loop;
	Fire:
		NAPF ABCDEFGHIJKLM 2;
		NAPR ABCDEFGHIJKLMNOPQRSTU 3;
		Goto Ready;
	AltFire:
		NASF ABCDEFGHIJ 3;
		NAPI A 4;
		NAP2 ABCDEFGHIJKL 3;
		Goto Ready;
	Spawn:
		TNT1 A 0;
		PICK A -1;
		Stop;
	}
}
