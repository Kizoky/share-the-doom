class PostNapalmLauncher : PostalWeapon
{
	Default
	{
		//$Category STD Weapons
		//$Title "Napalm Launcher"
		//$Sprite PICKA0
		Weapon.Kickback 100;
		Weapon.SelectionOrder 1300;
		Obituary "$OB_MPSHOTGUN";
		Tag "$TAG_SHOTGUN";
		Inventory.PickupMessage "$GOTSHOTGUN";
		+Weapon.NoAlert;
	}
	States
	{
	Ready:
		TNT1 A 1
		{
			cvar idlinganimcvar = cvar.findcvar('postal_idleanim');
			bool idlinganim = (idlingAnimCVar != null && idlingAnimCVar.GetBool ());
			
			if (idlinganim)
				return ResolveState ("Ready.IdleAnim");
			else
				return ResolveState ("Ready.NoIdleAnim");
			
			return ResolveState (null);
		}
		Loop;
	Ready.IdleAnim:
		NAPI ABCDEFGHIJKLMNOPQRS 7 A_WeaponReady;
		Goto Ready;
	Ready.NoIdleAnim:
		NAPI A 1 A_WeaponReady;
		Goto Ready;
	Deselect:
		TNT1 AAAAAAAAAAAAAAA 0 A_Lower;
		NAPD ABCDEFGHI 2;
		Loop;
	Select:
		TNT1 AAAAAAAAAAAAAAA 0 A_Raise;
		TNT1 A 0 A_WeaponOffset(0,32); // Fix "bump" when weapon is selected
		TNT1 A 0 A_PlaySound("weaponselect1", 7 | 4096);
		NAPS ABCDEFGHIJKLMNOPQ 2 A_WeaponReady;
		Loop;
	Fire:
		NAPF ABCDEFGHIJKLM 2;
		NAPR ABCDEFGHIJKLMNOPQRSTU 3;
		Goto Ready;
	AltFire:
		NASF ABCDEFGHIJ 3;
		NAPI A 4;
		NAP2 ABCDEFGHIJKL 3;
		Goto Ready;
	Spawn:
		TNT1 A 0;
		PICK A -1;
		Stop;
	}
}
