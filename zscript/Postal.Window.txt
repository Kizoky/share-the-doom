// Doom-ified/ZScriptifed version of Postal 2's Window
// Reinvented wheel? Probably not :)

class PostalWindow : Actor
{
	override int DamageMobj(Actor inflictor, Actor source, int damage, Name mod, int flags, double angle)
	{
		Destroy();
		return Super.DamageMobj(inflictor,source,damage,mod,flags,angle);
	}
	
	override void PostBeginPlay()
	{
		Super.PostBeginPlay();
	}
	
	override void Tick()
	{
		Super.Tick();
		if (level.time > TimeBeforeDestroy && TouchedOnce == true)
			Destroy();
	}
	
	int TimeBeforeDestroy;
	bool TouchedOnce;
	override void Touch (actor toucher)
	{
		if (toucher)
		{
			if (TouchedOnce) return;
			
			// There's a slight "bump" in Postal 2 when you touch the Window
			// Let's assume the toucher is falling
			if (toucher.pos.z > floorz)
			{
				TimeBeforeDestroy = level.time + 1;
				TouchedOnce = true;
			}
		}
	}
	
	Default
	{
		//$Category Gameplay
		//$Title "Window"
		//$Color 5
		Radius 5;
		Height 85;
		Speed 0;
		+SHOOTABLE
		+ROLLSPRITE
		+ABSMASKPITCH
		+MOVEWITHSECTOR
		+SOLID
		+SPECIAL
		BloodType "null";
		Renderstyle "Translucent";
		Alpha 0.68;
		PainChance 0;
		Health 666;
	}
	States
	{
	Spawn:
		PROP A 10;
		Loop;
	}
}
