class barrelcactus : PostalDecoration
{
	override void Touch(actor toucher)
	{
		if (toucher)
		{
			if (toucher is 'PostalCowhead' || toucher is 'PostalGrenade') return;
			
			int dd = toucher.DamageMobj(self, self, 1, "Melee");
			toucher.vel.xy = (0,0);
			toucher.A_Recoil(10);
			toucher.vel.z += 1;
		}
	
	}
	
	override void Tick()
	{
		Super.Tick();
	}
	
	Default
	{
		//$Category Plants
		//$Title "Cacti"
		Radius 20;
		Height 52;
		+SPECIAL
		+SOLID
		-NOBLOCKMAP
		-NOINTERACTION
		+SHOOTABLE
		
		Obituary "%o hugged a cactus to death.";
	}
	States
	{
	Spawn:
		PROP A -1;
		Loop;
	}
}
