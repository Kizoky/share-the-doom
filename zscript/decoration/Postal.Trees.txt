class Tree1 : PostalDecoration
{
	Default
	{
		//$Category Trees
		//$Title "Tree 1"
		Radius 40;
		Height 300;
		+SOLID
	}
	States
	{
	Spawn:
	    PROP A 1 Nodelay A_SpawnItemEx("Tree1_BoxCollider");
		PROP A -1;
		Stop;
	}
}


class Tree1_BoxCollider : Actor
{
	Default
	{
		Radius 40;
		Height 300;
		+SOLID
	}
	States
	{
	Spawn:
		TNT1 A 10;
		Loop;
	}
}
