class PostalKarmaActor : Actor 
{
	override void Tick() { }
	
	Default
	{
		+ROLLSPRITE
		+ABSMASKPITCH
		+ABSMASKANGLE
		+SHOOTABLE
		+BOUNCEONWALLS
		+BOUNCEONFLOORS
		+BOUNCEONCEILINGS
		+DONTBOUNCEONSKY
		+USEBOUNCESTATE
		+BOUNCEONACTORS
		+ALLOWBOUNCEONACTORS
		BounceFactor 0.95;
		//+NOGRAVITY
		-SOLID
		//+NOINTERACTION
	}
}

class trashcan1 : PostalKarmaActor
{
	Default
	{
		//$Category Physics Objects
		//$Title "Bin 1"
		Radius 10;
		Height 17;
	}
	States
	{
	Spawn:
		PROP A -1;
		Loop;
	}
}
