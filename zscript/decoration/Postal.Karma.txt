class PostalKarmaActor : Actor 
{
	override void Tick() { }
	
	Default
	{
		+ROLLSPRITE
		+ABSMASKPITCH
		+ABSMASKANGLE

		//+SOLID
		//+NOINTERACTION
	}
}
