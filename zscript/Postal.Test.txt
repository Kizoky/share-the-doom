class GiantKrotchyToy : Actor
{
	
	override void Tick()
	{
		Super.Tick();
	
	}
	
	Default
	{
		Health 50;
		Radius 1;
		Height 1;
		Speed 5;
		-SOLID
		MONSTER;
		SeeSound "krotchytoy/alert";
		Obituary "%k felt the wrath of the Krotchy-toy!";
	}
	States
	{
	Spawn:
		KROC AAAAAAAAAAAA 3 A_Look;
		Loop;
	See:
		MADD A 10;
		TNT1 A 0
		{
			bool randomm = random[krtchy](0,1);
			
			if (randomm)
				SetStateLabel("Leap");
			else SetStateLabel("See");
		
		}
		Loop;
	Leap:
		MADD A 2
		{
			A_FaceTarget();
			A_Recoil(-0.45);
			SetOrigin(pos+(0,0,18), TRUE);
			A_Recoil(-0.45);
			Vel.Z = (target.pos.Z + target.Height/2 - pos.Z) / DistanceBySpeed(target, 4);
		}
		TNT1 A 0 A_PlaySound("krotchytoy/boop",0);
		MADD A 5;
		Goto See;
}

}
