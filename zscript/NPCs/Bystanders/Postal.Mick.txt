class NPCMick : NPCCore
{
	override void NPC_Settings()
	{
	
		NPC_Scream = "dev/scream";
		NPC_Bump = "dev/bump";
		NPC_PetitionDeny = "noway";
		NPC_Dickout = "dev/dickout";
		NPC_Taunt = "dev/taunt";
		NPC_Panic = "dev/panic";
		NPC_Laugh = "dev/laugh";
		

		WalkingStyle = random[mystyle](1,6);
		

		MyVoice = frandom[voicef](0.95,1.12);
		

		class<actor> Head = "NPCHead";
		Actor MyHead = spawn(Head, self.pos, ALLOW_REPLACE);
		if (MyHead)
		{
			MyHead.Angle = self.Angle;
			let p = NPCHead(MyHead);
			if (p)
			{
				p.Owner = self;
				HeadHitbox = p;
			}
		}
		
		Hostility = "Hostile";
		NPCWeapon = "Pistol";
		
	}
	
	Default
	{
		//$Category NPCs
		//$Title "WJI Mick"
		//$Sprite DUDEA0
		Health 150;
		Radius 12;
		Height 65;
		AttackSound "";
		PainSound "dev/pain";
		DeathSound "dev/pain";
		Obituary "%o should replace this text of string.";
		CameraHeight 59;
		
		// Set this to enable custom Hostility and Weapon
		NPCCore.CustomBehavior true;
	}
}
